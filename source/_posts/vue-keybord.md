按键操作：
```javascript
function onKeyUp(obj){
      var count = amount.count;
      var num = $(obj).find('span').text();
      if(num == 'c'){
          amount.count = 0; 
      }else if(num == '.'){
          amount.count = count + num;
      }else{
         if(count === 0||count === '0'){
             amount.count = num;
         }else{
             amount.count = count + num;
         }
      }
      
}
function onDelete(){
      var count = amount.count.toString();
      if(count.length == 1) {
          amount.count = 0;
      }else{
          amount.count = count.substr(0,count.length-1)
      }
}
```
